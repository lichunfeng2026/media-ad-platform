            function updatePreview() {
                syncStateFromDOM();
                const uniqueId = 'preview_ad';
                const { innerHTML, containerStyle, isOverlayType } = buildAdInnerHTML(uniqueId, true);
                const s = state;

                // 更新广告预览槽
                const slotStyle = [];
                if (!isOverlayType) {
                    if (s.responsiveMode) {
                        slotStyle.push(`width:100%;max-width:${s.width}px;`);
                    } else {
                        slotStyle.push(`width:${s.width}px;`);
                    }
                    slotStyle.push('margin:10px auto;');
                } else if (s.adType === 'sticky-bottom') {
                    slotStyle.push('width:100%;position:sticky;bottom:0;z-index:50;');
                } else if (s.adType === 'popup') {
                    slotStyle.push('width:85%;max-width:480px;margin:0 auto;');
                } else if (s.adType === 'interstitial') {
                    slotStyle.push('width:88%;max-width:560px;margin:0 auto;');
                }
                slotStyle.push(`background:${s.bgColor};`);
                slotStyle.push(`border-radius:${s.borderRadius}px;`);
                slotStyle.push(`box-shadow:${getShadowCSS(s.shadowLevel)};`);
                slotStyle.push('overflow:hidden;');
                slotStyle.push('font-family:inherit;');
                slotStyle.push('box-sizing:border-box;');

                adPreviewSlot.style.cssText = slotStyle.join(';');
                adPreviewSlot.innerHTML = innerHTML;

                // 处理覆盖层
                if (['popup', 'interstitial'].includes(s.adType)) {
                    mockOverlay.style.display = 'flex';
                    mockOverlay.style.background = `rgba(0,0,0,${s.overlayOpacity.toFixed(2)})`;
                    // 把广告移到覆盖层内显示
                    if (adPreviewSlot.parentElement === mockPageContent) {
                        mockOverlay.appendChild(adPreviewSlot);
                    }
                    adPreviewSlot.style.position = 'relative';
                    adPreviewSlot.style.zIndex = '1';
                    adPreviewSlot.style.maxWidth = s.adType === 'interstitial' ? '88%' : '85%';
                } else if (s.adType === 'sticky-bottom') {
                    if (adPreviewSlot.parentElement === mockOverlay) {
                        mockPageContent.appendChild(adPreviewSlot);
                    }
                    mockOverlay.style.display = 'none';
                    adPreviewSlot.style.position = 'sticky';
                    adPreviewSlot.style.bottom = '0';
                    adPreviewSlot.style.zIndex = '50';
                    adPreviewSlot.style.width = '100%';
                } else {
                    if (adPreviewSlot.parentElement === mockOverlay) {
                        mockPageContent.appendChild(adPreviewSlot);
                    }
                    mockOverlay.style.display = 'none';
                    adPreviewSlot.style.position = 'relative';
                    adPreviewSlot.style.zIndex = '5';

                    if (innerEl) {
                        innerEl.style.animation = 'none';
                        requestAnimationFrame(() => {
                            innerEl.style.animation = getAnimationName(s.animationType);
                        });
                    }
                }
            }

            function copyToClipboard(text, toastMsg) {
                if (navigator.clipboard && navigator.clipboard.writeText) {
                    navigator.clipboard.writeText(text).then(() => showToast(toastMsg)).catch(() => {
                        fallbackCopy(text, toastMsg);
                    });
                } else {
                    fallbackCopy(text, toastMsg);
                }
            }

            function fallbackCopy(text, toastMsg) {
                const ta = document.createElement('textarea');
                ta.value = text;
                ta.style.cssText = 'position:fixed;left:-9999px;top:-9999px;';
                document.body.appendChild(ta);
                ta.select();
                try { document.execCommand('copy');
                    showToast(toastMsg); } catch (e) { showToast('复制失败，请手动复制'); }
                document.body.removeChild(ta);
            }

            function downloadAsFile(content, filename) {
                const blob = new Blob([content], { type: 'text/html;charset=utf-8' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = filename;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
                showToast('文件下载成功！可上传到任意网站使用');
            }

            function showToast(msg) {
                const existing = document.querySelector('.toast');
                if (existing) existing.remove();
                const toast = document.createElement('div');
                toast.className = 'toast';
                toast.textContent = msg;
                document.body.appendChild(toast);
                setTimeout(() => toast.remove(), 2300);
            }
